%% 需要同学生成调制信号、载波信号以及已调信号
fm = 5; fc = 25; kf = 10;
A = 5; A_c = 1;
mt = A * cos(2 * pi * fm * t);
ct = cos(2 * pi * fc * t);
sFM = A_c * cos(2 * pi * fc * t + kf * cumsum(mt) * dt);
