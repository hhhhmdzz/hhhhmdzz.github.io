r2a = lambda r: r * 180 / np.pi

运动学：

$v$ 车辆的实际速度

$\varphi$ 车辆的航偏角

$x, y$ 车辆横向与纵向的位置

$L$ 轴距

$R$ 转弯半径

$\delta$ 车辆的前轮偏角

$\dot{\varphi} = \omega = \frac{v}{R} \overset{tan\delta = \frac{L}{R}}{=} v * \frac{tan\delta}{L}$

$\begin{align} \dot{x} & = v * cos\varphi\\ \dot{y} & = v * sin\varphi\\ \dot{\varphi} & = v * \frac{tan\delta}{L} \end{align} \Rightarrow \left[\begin{matrix} \dot{x}\\ \dot{y}\\ \dot{\varphi}\\ \end{matrix}\right] = \left[\begin{matrix} cos\varphi\\ sin\varphi\\ \frac{tan\delta}{l}\\ \end{matrix}\right] v$

状态转移方程：

状态量：$X = \left(\begin{matrix} x,y,\varphi \end{matrix}\right)$

控制量：$u = \left(\begin{matrix} v,\delta \end{matrix}\right)$

$\bar{X} = X - X_r$

$\begin{align} \nonumber \bar{X}(k+1) & = & A_{k, t} & \bar{X}(k) + & B_{k, t}\bar{u}(k)\\ \nonumber & = & \left[\begin{matrix} 1 & 0 & {-v_rsin\varphi_rT_s}\\ 0 & 1 & {v_rcos\varphi_rT_s}\\ 0 & 0 & 1\\ \end{matrix}\right] & \bar{X}(k) + & \left[\begin{matrix} {cos\varphi_rT_s} & 0 \\ {sin\varphi_rT_s} & 0 \\ \cfrac{tan\delta_rT_s}{l} & \cfrac{v_rT_s}{lcos^2\delta_r}\\ \end{matrix}\right]\bar{u}(k)\\ \end{align}$

$\delta(k+1) = \omega \Delta T + \delta(k)$

def judge_vertical(delta, varphi, approach_radian, R, L=L, to_print=0):
    """
    :param delta: 弧度
    :param varphi: 弧度
    :param approach_radian: 逼近弧度

    lr = lf = L / 2  # 轴距

    delta_r = 0  # 后轮

def fA(v, varphi, Ts=Ts):
    return np.array([
        [1, 0, - v * np.sin(varphi) * Ts],
        [0, 1, v * np.cos(varphi) * Ts],
        [0, 0, 1],
    ])
def fB(varphi, delta, l=L, Ts=Ts, v=v):
    return np.array([
        [np.cos(varphi) * Ts, 0],
        [np.sin(varphi) * Ts, 0],
        [np.tan(delta) * Ts / l, v * Ts / l / np.cos(delta)**2],
    ])
def fu(v, delta):
    return np.array([v, delta])
